#!/usr/bin/python


def ddd(aaa):
	"""To read num of lines,num of words"""
	num_lines=0
	num_words=0
	with open(aaa,'r+') as input_file:
		for line in input_file:
			num_lines+=1                      #reads number of lines
			line_words=line.strip().split()
			num_words+=len(line_words)        #reads number of words
	print "line count:{0},word count:{1}".format(num_lines,num_words)
ddd('1.txt')
